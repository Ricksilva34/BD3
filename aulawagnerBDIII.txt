-Criar a tabela Clientes
CREATE TABLE Pedidos (
    id INT PRIMARY KEY AUTO_INCREMENT,
    data_pedido DATE NOT NULL,
    valor DECIMAL(10, 2) NOT NULL,
    cliente_id INT,
    FOREIGN KEY (cliente_id) REFERENCES Clientes(id)
);

-Criar a tabela Pedidos
CREATE TABLE Pedidos (
    id INT PRIMARY KEY AUTO_INCREMENT,
    data_pedido DATE NOT NULL,
    valor DECIMAL(10, 2) NOT NULL,
    cliente_id INT,
    FOREIGN KEY (cliente_id) REFERENCES Clientes(id)
);
